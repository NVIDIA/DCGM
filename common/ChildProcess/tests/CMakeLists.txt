    return()
endif()

add_executable(childprocesstests)
target_sources(childprocesstests PRIVATE
    FramedChannelTests.cpp
    ChildProcessBuilderTests.cpp
    StdLinesTests.cpp
    PipeTests.cpp
)

target_link_libraries(childprocesstests
    PRIVATE
    childprocess
    Catch2::Catch2WithMain
    ${CMAKE_THREAD_LIBS_INIT}
    fmt::fmt
    rt
)

add_executable(childprocesstesttool)
target_sources(childprocesstesttool PRIVATE
    ChildProcessTestTool.cpp
)

if (CMAKE_SYSTEM_PROCESSOR STREQUAL "x86_64")
    catch_discover_tests(childprocesstests EXTRA_ARGS --colour-mode ansi)
endif ()
